[ 1. qi dong fu wu qi ]

  use HTTP::Handy;

  my $app = sub {
      my $env = shift;
      return [200, ['Content-Type', 'text/plain'], ['Hello']];
  };

  HTTP::Handy->run(
      app           => $app,       # bi tian: PSGI app code ref
      host          => '127.0.0.1',# ke xuan: bind address (0.0.0.0)
      port          => 8080,       # ke xuan: port (yu she: 8080)
      log           => 1,          # ke xuan: access log (yu she: 1)
      max_post_size => 10485760,   # ke xuan: max POST bytes (yu she: 10MB)
  );

  zhi jie zuo wei jiao ben zhi xing: perl lib/HTTP/Handy.pm [port]
